ALT_study_effect = {
	duel = {
		skill = learning
		value = 40
		5 = {
			compare_modifier = {
				value = scope:duel_value
				multiplier = 1
			}
			modifier = {
				var:current_pursuit = { has_variable = se_success_chance }
				add = {
					value = var:current_pursuit.var:se_success_chance
					multiply = 15
				}
			}
			if = { limit = { var:current_pursuit = { has_variable = se_success_chance } } custom_tooltip = ALT_study_effect_tt }
			send_interface_toast = {
				title = board_games.0001.tt.success
				ALT_school_xp_gain = { SCHOOL = $SCHOOL$ COUNT = ALT_xp_gain }
			}
		}
		5 = {
			compare_modifier = {
				value = scope:duel_value
				multiplier = -1
			}
			send_interface_toast = {
				title = board_games.0001.tt.failure
			}
		}
	}
}

	duel = {
		skill = learning
		value = 60
		5 = {
			compare_modifier = {
				value = scope:duel_value
				multiplier = 1
			}
			modifier = {
				trigger_if = { limit = { has_variable = current_pursuit } var:current_pursuit = { has_variable = se_success_chance } }
				trigger_else = { always = no }
				add = {
					value = var:current_pursuit.var:se_success_chance
					multiply = 15
				}

				title = board_games.0001.tt.success

					1 = { trigger = { NOT = { has_character_flag = ALT_know_biomancy } } ALT_become_biomancer = yes }

					1 = { trigger = { NOT = { has_character_flag = ALT_know_aetherimancy } } ALT_become_aetherimancer = yes }
